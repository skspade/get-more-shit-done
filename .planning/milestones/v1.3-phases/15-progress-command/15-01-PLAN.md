  <action>
  Replace the handleProgress stub with a real implementation. The function signature is `handleProgress(projectRoot, args, mode)`.

  **Step 1: Add requires at top of cli.cjs.**
  Add imports for the modules needed:
  ```javascript
  const { getMilestoneInfo } = require('./core.cjs');
  const { extractFrontmatter } = require('./frontmatter.cjs');
  const { cmdRoadmapAnalyze } = require('./roadmap.cjs');
  ```

  **Step 2: Create a helper to capture roadmap analyze output.**
  `cmdRoadmapAnalyze` calls `output()` internally (it's designed for gsd-tools CLI). We need its data directly. Instead of calling it, replicate the data gathering logic inline or call the internal functions. The simplest approach: read the data the same way cmdRoadmapAnalyze does, but return it directly. Extract the phase-reading logic into a helper function `gatherProgressData(projectRoot)` that returns the structured data object.

  The `gatherProgressData(projectRoot)` function should:
  1. Call `getMilestoneInfo(projectRoot)` to get `{ version, name }`
  2. Read `.planning/STATE.md` and extract frontmatter to get milestone status. If frontmatter has `status` field, use it. Default to 'active' if not found or if status is 'unknown'.
  3. Read `.planning/phases/` directory. For each phase subdirectory (sorted by phase number using the pattern from cmdProgressRender):
     - Count files ending in `-PLAN.md` or `PLAN.md` as plans
     - Count files ending in `-SUMMARY.md` or `SUMMARY.md` as summaries
     - Determine status: 'complete' if summaries >= plans and plans > 0, 'in_progress' if summaries > 0 but < plans, 'planned' if plans > 0 but summaries == 0, 'not_started' otherwise
     - Extract phase number and name from directory name pattern `{number}-{name}`
  4. Compute totals: totalPlans, completedPlans (totalSummaries), percent
  5. Determine current position: first phase that is 'in_progress', or first 'planned' phase, or first 'not_started' phase
  6. Determine next action based on current phase lifecycle state:
     - If no phases exist: "No phases found"
     - If all phases complete: "All phases complete -- milestone done"
     - If current phase has no plans: "Plan phase {N}: gsd plan-phase {N}" (note: we check has_context via presence of CONTEXT.md files in the directory)
     - If current phase has plans but incomplete: "Execute phase {N}: gsd execute-phase {N}"
     - If current phase plans all complete: "Verify phase {N}: gsd verify-phase {N}"

  Return structure:
  ```javascript
  {
    milestone: { name, version, status },
    phases: [{ number, name, status_indicator, plan_counts }],
    progress: { percent, completed_plans, total_plans, bar },
    current_position: { phase, plan },
    next_action: "string"
  }
  ```

  Where:
  - `status_indicator` is one of: 'complete', 'in_progress', 'planned', 'not_started'
  - `plan_counts` is a string like "2/3 plans"
  - `bar` is a string using block characters: `'\u2588'.repeat(filled) + '\u2591'.repeat(barWidth - filled)` with barWidth of 20
  - `current_position.phase` is the phase number string, `current_position.plan` is null (plan-level tracking not available from disk)

  **Step 3: Implement handleProgress(projectRoot, args, mode).**

  Call `gatherProgressData(projectRoot)` to get the data.

  If mode is 'json', return the data object directly (formatOutput will JSON.stringify it).

  For 'rich' and 'plain' modes, build a formatted string with ANSI colors (plain mode stripping happens in formatOutput):

  ```
  {bold}{milestone.name} {milestone.version}{reset} {status_color}({milestone.status}){reset}

  Phases:
    {status_icon} Phase {number}: {name}  {plan_counts}
    {status_icon} Phase {number}: {name}  {plan_counts}
    ...

  Progress: [{bar}] {completed_plans}/{total_plans} plans ({percent}%)

  Current: Phase {current_position.phase}
  Next: {next_action}
  ```

  ANSI codes to use:
  - Bold: `\x1b[1m`, Reset: `\x1b[0m`
  - Green: `\x1b[32m` (for complete status and checkmark icon)
  - Yellow: `\x1b[33m` (for in_progress status and arrow icon)
  - Dim/gray: `\x1b[2m` (for not_started status and dash icon)
  - Cyan: `\x1b[36m` (for planned status)

  Status icons:
  - complete: green checkmark (e.g., `\x1b[32m\u2713\x1b[0m`)
  - in_progress: yellow arrow (e.g., `\x1b[33m\u25B6\x1b[0m`)
  - planned: cyan circle (e.g., `\x1b[36m\u25CB\x1b[0m`)
  - not_started: dim dash (e.g., `\x1b[2m-\x1b[0m`)

  Return `{ command: 'progress', ...data, message: formattedString }` so formatOutput handles all three modes correctly:
  - json mode: returns the full object (with milestone, phases, progress, etc.)
  - plain mode: extracts `message` and strips ANSI
  - rich mode: extracts `message` and passes through

  describe('handleProgress', () => {
    let tmpDir;

    beforeEach(() => {
      tmpDir = fs.mkdtempSync(path.join(os.tmpdir(), 'gsd-progress-test-'));
      // Create .planning structure
      const planningDir = path.join(tmpDir, '.planning');
      fs.mkdirSync(path.join(planningDir, 'phases', '01-setup'), { recursive: true });
      fs.mkdirSync(path.join(planningDir, 'phases', '02-feature'), { recursive: true });
      // Create ROADMAP.md with milestone info
      fs.writeFileSync(path.join(planningDir, 'ROADMAP.md'),
        '## v1.0 Test Milestone\n\n- [ ] Phase 1: Setup\n- [ ] Phase 2: Feature\n');
      // Create STATE.md with frontmatter
      fs.writeFileSync(path.join(planningDir, 'STATE.md'),
        '---\nmilestone: v1.0\nmilestone_name: Test Milestone\nstatus: active\n---\n');
      // Phase 1: 2 plans, 2 summaries (complete)
      fs.writeFileSync(path.join(planningDir, 'phases', '01-setup', '01-01-PLAN.md'), '---\nplan: 01\n---\n');
      fs.writeFileSync(path.join(planningDir, 'phases', '01-setup', '01-02-PLAN.md'), '---\nplan: 02\n---\n');
      fs.writeFileSync(path.join(planningDir, 'phases', '01-setup', '01-01-SUMMARY.md'), 'done');
      fs.writeFileSync(path.join(planningDir, 'phases', '01-setup', '01-02-SUMMARY.md'), 'done');
      // Phase 2: 2 plans, 1 summary (in progress)
      fs.writeFileSync(path.join(planningDir, 'phases', '02-feature', '02-01-PLAN.md'), '---\nplan: 01\n---\n');
      fs.writeFileSync(path.join(planningDir, 'phases', '02-feature', '02-02-PLAN.md'), '---\nplan: 02\n---\n');
      fs.writeFileSync(path.join(planningDir, 'phases', '02-feature', '02-01-SUMMARY.md'), 'done');
    });

    afterEach(() => {
      fs.rmSync(tmpDir, { recursive: true, force: true });
    });

    test('json mode returns milestone info (PROG-01)', () => {
      const result = routeCommand('progress', tmpDir, [], 'json');
      assert.strictEqual(result.milestone.version, 'v1.0');
      assert.strictEqual(result.milestone.name, 'Test Milestone');
      assert.ok(['active', 'unknown'].includes(result.milestone.status));
    });

    test('json mode returns phases with status indicators (PROG-02)', () => {
      const result = routeCommand('progress', tmpDir, [], 'json');
      assert.ok(Array.isArray(result.phases));
      assert.ok(result.phases.length >= 2);
      const phase1 = result.phases.find(p => p.number === '01' || p.number === '1');
      const phase2 = result.phases.find(p => p.number === '02' || p.number === '2');
      assert.strictEqual(phase1.status_indicator, 'complete');
      assert.strictEqual(phase2.status_indicator, 'in_progress');
    });

    test('json mode returns plan counts per phase (PROG-03)', () => {
      const result = routeCommand('progress', tmpDir, [], 'json');
      const phase1 = result.phases.find(p => p.number === '01' || p.number === '1');
      assert.ok(phase1.plan_counts.includes('2/2'));
    });

    test('json mode returns progress bar data (PROG-04)', () => {
      const result = routeCommand('progress', tmpDir, [], 'json');
      assert.strictEqual(typeof result.progress.percent, 'number');
      assert.strictEqual(result.progress.total_plans, 4);
      assert.strictEqual(result.progress.completed_plans, 3);
      assert.ok(result.progress.bar.length > 0);
    });

    test('json mode returns current position and next action (PROG-05)', () => {
      const result = routeCommand('progress', tmpDir, [], 'json');
      assert.ok(result.current_position);
      assert.ok(result.next_action);
      assert.strictEqual(typeof result.next_action, 'string');
    });

    test('rich mode returns message with ANSI codes', () => {

      assert.ok(result.message);
      assert.ok(result.message.includes('\x1b['));

        fs.rmSync(emptyDir, { recursive: true, force: true });
      }
    });
  });
