2. **Config loading** — read circuit breaker threshold from CJS config:
   ```js
   function getConfig(key, defaultVal) {
     try {
       // Try config.json first, fall back to CONFIG_DEFAULTS
       const configPath = path.join(PROJECT_DIR, '.planning', 'config.json');
       if (fs.existsSync(configPath)) {
         const config = JSON.parse(fs.readFileSync(configPath, 'utf-8'));
         const keys = key.split('.');
         let val = config;
         for (const k of keys) {
           if (val === undefined || val === null) break;
           val = val[k];
         }
         if (val !== undefined && val !== null) return val;
       }
     } catch {}
     // Fall back to CONFIG_DEFAULTS from config.cjs
     if (CONFIG_DEFAULTS[key] !== undefined) return CONFIG_DEFAULTS[key];
     return defaultVal;
   }

   const CIRCUIT_BREAKER_THRESHOLD = getConfig('autopilot.circuit_breaker_threshold', 3);
   ```

3. **`takeProgressSnapshot()`** — matches bash signals exactly:
   ```js
   async function takeProgressSnapshot() {
     const commits = await $`cd ${PROJECT_DIR} && git rev-list --count HEAD`.nothrow().quiet();
     const commitCount = commits.stdout.trim() || '0';
     const artifactFiles = await $`find ${path.join(PROJECT_DIR, '.planning/phases')} -name "*.md" 2>/dev/null | wc -l`.nothrow().quiet();
     const artifactCount = artifactFiles.stdout.trim().replace(/\s/g, '') || '0';
     return `${commitCount}|${artifactCount}`;
   }
   ```

4. **`checkProgress(before, after, stepDesc)`** — matches bash logic:
   - Increments `totalIterations`
   - Logs: `PROGRESS CHECK: step={stepDesc} before={before} after={after}`
   - If before === after: increment `noProgressCount`, add to `iterationLog`, log "NO PROGRESS", check threshold
   - If threshold reached: log "CIRCUIT BREAKER TRIGGERED", call `printHaltReport`, `process.exit(1)`
   - Print warning to stderr: `WARNING: No progress detected (N/THRESHOLD consecutive)`
   - If before !== after: reset `noProgressCount = 0`, clear `iterationLog = []`, log "PROGRESS DETECTED"

5. **`printHaltReport(reason, step, exitCode)`** — matches bash box format exactly:

   ```
   ║  AUTOPILOT HALTED                                           ║
   ╚══════════════════════════════════════════════════════════════╝

   Reason: {reason}
   Phase: {CURRENT_PHASE}
   Step: {step}
   Exit code: {exitCode}
   Total iterations: {totalIterations}
   Consecutive no-progress: {noProgressCount}

   Last iterations:
     - {each iterationLog entry}

   Progress signals checked:
     - Git commit count
     - Artifact file count in .planning/phases/

   To resume: autopilot.sh --from-phase {CURRENT_PHASE} --project-dir {PROJECT_DIR}
   Log file: {LOG_FILE}
   ```
   Also logs all entries to LOG_FILE matching bash format.

Add the `runStep` function that spawns `claude -p` via zx. This is the simple step runner (Phase 48 scope). Debug retry (`runStepWithRetry`) is Phase 49 scope.

1. **`runStep(prompt, stepName)`** — ports bash `run_step`:
   ```js
   async function runStep(prompt, stepName) {
     printBanner(`Phase ${CURRENT_PHASE} > ${stepName}`);

     const snapshotBefore = await takeProgressSnapshot();

     if (DRY_RUN) {
       logMsg(`STEP DRY-RUN: ${stepName}`);
       console.log(`[DRY RUN] Would execute:`);
       console.log(`  claude -p --dangerously-skip-permissions --output-format json "${prompt}"`);
       console.log('');
       checkProgress(snapshotBefore, snapshotBefore, `${stepName} (dry-run)`);
       return 0;
     }

     logMsg(`STEP START: phase=${CURRENT_PHASE} step=${stepName}`);

     const result = await $`cd ${PROJECT_DIR} && claude -p --dangerously-skip-permissions --output-format json ${prompt}`.nothrow();

     const exitCode = result.exitCode;
     logMsg(`STEP DONE: step=${stepName} exit_code=${exitCode}`);

     // Print stdout for transparency
     if (result.stdout) process.stdout.write(result.stdout);

     const snapshotAfter = await takeProgressSnapshot();
     checkProgress(snapshotBefore, snapshotAfter, stepName);

     if (exitCode !== 0) {
       if (snapshotBefore === snapshotAfter) {
         console.error(`ERROR: Step '${stepName}' failed with exit code ${exitCode} and no artifacts created`);
         printHaltReport('Step failure with no progress', stepName, exitCode);
         process.exit(1);
       } else {
         console.error(`WARNING: Step '${stepName}' exited with code ${exitCode} but made progress. Continuing.`);
       }
     }

     return exitCode;
   }
   ```

   Key behaviors matching bash:
   - Takes progress snapshot before and after
   - Dry-run prints but does not spawn, still checks progress (triggers circuit breaker)
   - Uses `.nothrow()` so non-zero exit doesn't throw
   - On failure with no progress: halt report + exit 1
   - On failure with progress: warning, continue (partial success)
