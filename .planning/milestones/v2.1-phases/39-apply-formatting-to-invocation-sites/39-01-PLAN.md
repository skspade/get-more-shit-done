# Pretty-print JSON output; pass through non-JSON unchanged
format_json_output() {
  local input
  input=$(cat)
  if echo "$input" | jq . >/dev/null 2>&1; then
    echo "$input" | jq .
  else
    echo "$input"
  fi
}
